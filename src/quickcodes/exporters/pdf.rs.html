1//! PDF export functionality for barcodes
2//!
3//! This module provides native PDF export capabilities using the printpdf crate.
4//! Supports both 1D and 2D barcodes with high-quality vector output.
5//!
6//! Note: This is a simplified implementation for demonstration purposes.
7//! A production implementation would use more advanced PDF rendering techniques.
8
9use crate::types::{Barcode, BarcodeModules, QuickCodesError, Result};
10
11/// Export a barcode to PDF format
12///
13/// This is a simplified implementation that generates a basic PDF
14/// with the barcode data as text. A full implementation would render
15/// the actual barcode graphics.
16///
17/// # Arguments
18/// * `barcode` - The barcode to export
19///
20/// # Returns
21/// Returns the PDF data as bytes
22pub fn export_pdf(barcode: &Barcode) -> Result<Vec<u8>> {
23    // For now, create a simple PDF with the barcode information
24    // This is a placeholder implementation
25    generate_simple_pdf(barcode)
26}
27
28/// Generate a simple PDF with barcode information
29fn generate_simple_pdf(barcode: &Barcode) -> Result<Vec<u8>> {
30    use printpdf::{BuiltinFont, Mm, PdfDocument};
31
32    // Create a new PDF document
33    let (doc, page1, layer1) =
34        PdfDocument::new("QuickCodes Barcode", Mm(210.0), Mm(297.0), "Layer 1");
35    let current_layer = doc.get_page(page1).get_layer(layer1);
36
37    // Add built-in font
38    let font = doc
39        .add_builtin_font(BuiltinFont::Helvetica)
40        .map_err(|e| QuickCodesError::ExportError(format!("Font error: {}", e)))?;
41
42    // Add barcode information as text
43    current_layer.begin_text_section();
44    current_layer.set_font(&font, 12.0);
45    current_layer.set_text_cursor(Mm(20.0), Mm(250.0));
46    current_layer.write_text(format!("Barcode Type: {:?}", barcode.barcode_type), &font);
47    current_layer.set_text_cursor(Mm(20.0), Mm(230.0));
48    current_layer.write_text(format!("Data: {}", barcode.data), &font);
49
50    // Add dimensions info
51    let dimensions = match &barcode.modules {
52        BarcodeModules::Linear(pattern) => format!("Pattern Length: {}", pattern.len()),
53        BarcodeModules::Matrix(matrix) => format!(
54            "Matrix Size: {}x{}",
55            matrix.len(),
56            matrix.first().map(|row| row.len()).unwrap_or(0)
57        ),
58    };
59    current_layer.set_text_cursor(Mm(20.0), Mm(210.0));
60    current_layer.write_text(dimensions, &font);
61
62    current_layer.end_text_section();
63
64    // Save to bytes
65    let mut buffer = Vec::new();
66    {
67        let mut writer = std::io::BufWriter::new(&mut buffer);
68        doc.save(&mut writer)
69            .map_err(|e| QuickCodesError::ExportError(format!("PDF save failed: {}", e)))?;
70    } // writer is dropped here, releasing the borrow
71
72    Ok(buffer)
73}

75#[cfg(test)]
76mod tests {
77    use super::*;
78    use crate::types::{BarcodeConfig, BarcodeType};
79
80    #[test]
81    #[cfg(feature = "pdf")]
82    fn test_pdf_export_linear() {
83        let barcode = Barcode {
84            barcode_type: BarcodeType::Code128,
85            data: "TEST123".to_string(),
86            modules: BarcodeModules::Linear(vec![
87                true, false, true, true, false, true, false, false,
88            ]),
89            config: BarcodeConfig::default(),
90        };
91
92        let result = export_pdf(&barcode);
93        assert!(result.is_ok());
94
95        let pdf_data = result.unwrap();
96        assert!(!pdf_data.is_empty());
97
98        // Check PDF header
99        assert!(pdf_data.starts_with(b"%PDF"));
100    }
