1//! PDF417 barcode generator

3//! PDF417 is a stacked linear barcode commonly used in official documents,
4//! driver's licenses, and identification cards.

10/// PDF417 configuration options
11#[derive(Debug, Clone)]
12pub struct PDF417Config {
13    /// Number of data columns (1-30)
14    pub columns: u8,
15    /// Error correction level (0-8)
16    pub error_correction: u8,
17    /// Compact format (reduces size)
18    pub compact: bool,
19}
20
21impl Default for PDF417Config {
22    fn default() -> Self {
23        Self {
24            columns: 6,
25            error_correction: 2,
26            compact: false,
27        }
28    }
29}

31/// Generate a PDF417 with default configuration
32pub fn generate_pdf417(data: &str) -> Result<Barcode> {
33    generate_pdf417_with_config(data, &BarcodeConfig::default())
34}

36/// Generate a PDF417 with custom configuration
37pub fn generate_pdf417_with_config(data: &str, config: &BarcodeConfig) -> Result<Barcode> {
38    if data.is_empty() {
39        return Err(QuickCodesError::InvalidData(
40            "PDF417 data cannot be empty".to_string(),
41        ));
42    }
43
44    // For now, implement a basic PDF417 structure
45    // A full implementation would require complex Reed-Solomon error correction
46    let pdf417_config = PDF417Config::default();
47    let matrix = generate_pdf417_matrix(data, &pdf417_config)?;
48
49    Ok(Barcode {
50        barcode_type: BarcodeType::PDF417,
51        data: data.to_string(),
52        modules: BarcodeModules::Matrix(matrix),
53        config: config.clone(),
54    })
55}

57/// Generate PDF417 matrix (simplified implementation)
58fn generate_pdf417_matrix(data: &str, config: &PDF417Config) -> Result<Vec<Vec<bool>>> {
59    // This is a simplified implementation for demonstration
60    // A full PDF417 implementation would need:
61    // 1. Data compaction (Text, Byte, Numeric modes)
62    // 2. Reed-Solomon error correction
63    // 3. Proper symbol construction with start/stop patterns
64    // 4. Row indicators and column indicators
65
66    let data_bytes = data.as_bytes();
67    let rows = calculate_rows(data_bytes.len(), config.columns as usize, config.error_correction);
68    let cols = config.columns as usize * 17 + 34; // Each data column is 17 modules + start/stop
69
70    let mut matrix = vec![vec![false; cols]; rows];
71
72    // Generate basic pattern (placeholder)
73    for (row_idx, row) in matrix.iter_mut().enumerate() {
74        // Start pattern (8 modules)
75        for i in 0..8 {
76            row[i] = (i + row_idx) % 2 == 0;
77        }
78
79        // Data area (simplified encoding)
80        let start_data = 8;
81        let end_data = cols - 8;
82        for (col_idx, cell) in row[start_data..end_data].iter_mut().enumerate() {
83            let byte_idx = (row_idx * (end_data - start_data) + col_idx) / 8;
84            let bit_idx = (row_idx * (end_data - start_data) + col_idx) % 8;
85            
86            if byte_idx < data_bytes.len() {
87                *cell = (data_bytes[byte_idx] >> (7 - bit_idx)) & 1 == 1;
88            } else {
89                *cell = false; // Padding
90            }
91        }
92
93        // Stop pattern (8 modules)
94        for i in 0..8 {
95            row[cols - 8 + i] = (i + row_idx + 1) % 2 == 0;
96        }
97    }
98
99    Ok(matrix)
100}

102/// Calculate number of rows needed for PDF417
103fn calculate_rows(data_len: usize, columns: usize, error_level: u8) -> usize {
104    // Simplified calculation
105    // Real PDF417 would use complex formulas based on data compaction
106    let error_codewords = match error_level {
107        0 => 2,
108        1 => 4,
109        2 => 8,
110        3 => 16,
111        4 => 32,
112        5 => 64,
113        6 => 128,
114        7 => 256,
115        8 => 512,
116        _ => 8,
117    };
118
119    let total_codewords = data_len + error_codewords + 1; // +1 for length indicator
120    let rows = (total_codewords + columns - 1) / columns; // Ceiling division
121    
122    // PDF417 must have 3-90 rows
123    rows.max(3).min(90)
124}

126#[cfg(test)]
127mod tests {
128    use super::*;
129
130    #[test]
131    fn test_pdf417_generation() {
132        let result = generate_pdf417("Hello PDF417");
133        assert!(result.is_ok());
134
135        let barcode = result.unwrap();
136        assert_eq!(barcode.barcode_type, BarcodeType::PDF417);
137        assert_eq!(barcode.data, "Hello PDF417");
138
139        match barcode.modules {
140            BarcodeModules::Matrix(matrix) => {
141                assert!(!matrix.is_empty());
142                assert!(!matrix[0].is_empty());
143                // PDF417 should have at least 3 rows
144                assert!(matrix.len() >= 3);
145            }
146            _ => panic!("PDF417 should generate a matrix pattern"),
147        }
148    }
149
150    #[test]
151    fn test_pdf417_empty_data() {
152        let result = generate_pdf417("");
153        assert!(result.is_err());
154    }
155
156    #[test]
157    fn test_pdf417_document_data() {
158        // Test with document-like data
159        let doc_data = "DRIVER LICENSE|DOE,JOHN|123456789|2025-12-31";
160        let result = generate_pdf417(doc_data);
161        assert!(result.is_ok());
162
163        let barcode = result.unwrap();
164        assert_eq!(barcode.data, doc_data);
165    }

167    #[test]
168    fn test_calculate_rows() {
169        // Test with actual values and verify they are in valid ranges
170        let result1 = calculate_rows(10, 6, 2);
171        let result2 = calculate_rows(100, 6, 2);
172        let result3 = calculate_rows(1000, 6, 2);
173        
174        // PDF417 should have 3-90 rows
175        assert!(result1 >= 3 && result1 <= 90, "Rows should be 3-90, got {}", result1);
176        assert!(result2 >= 3 && result2 <= 90, "Rows should be 3-90, got {}", result2);
177        assert!(result3 >= 3 && result3 <= 90, "Rows should be 3-90, got {}", result3);
178        
179        // Larger data should need more rows
180        assert!(result2 > result1, "More data should need more rows");
181        assert!(result3 >= result2, "Much more data should need at least as many rows");
182    }
