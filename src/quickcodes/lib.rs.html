20/// Main generation function - unified API for all barcode types
21///
22/// # Arguments
23///
24/// * `barcode_type` - The type of barcode to generate
25/// * `data` - The data to encode
26/// * `format` - The output format (PNG, SVG)
27///
28/// # Returns
29///
30/// Returns the generated barcode as bytes in the specified format
31///
32/// # Examples
33///
34/// ```rust
35/// use quickcodes::{generate, BarcodeType, ExportFormat};
36///
37/// fn main() -> Result<(), Box<dyn std::error::Error>> {
38///     // Generate QR Code as SVG
39///     let svg_data = generate(BarcodeType::QRCode, "https://example.com", ExportFormat::SVG)?;
40///
41///     // Generate EAN-13 as PNG
42///     let png_data = generate(BarcodeType::EAN13, "123456789012", ExportFormat::PNG)?;
43///     
44///     Ok(())
45/// }
46/// ```
47pub fn generate(
48    barcode_type: BarcodeType,
49    data: &str,
50    format: ExportFormat,
51) -> AnyhowResult<Vec<u8>> {
52    if data.is_empty() {
53        return Err(anyhow::anyhow!("Data cannot be empty"));
54    }
55
56    let barcode = match barcode_type {
57        // Phase 1: Core formats
58        BarcodeType::QRCode => generators::qr::generate_qr(data)?,
59        BarcodeType::EAN13 => generators::ean13::generate_ean13(data)?,
60        BarcodeType::UPCA => generators::upc::generate_upc_a(data)?,
61        BarcodeType::Code128 => generators::code128::generate_code128(data)?,
62
63        // Phase 2: Advanced 2D codes
64        BarcodeType::DataMatrix => generators::datamatrix::generate_datamatrix(data)?,
65        BarcodeType::PDF417 => generators::pdf417::generate_pdf417(data)?,
66        BarcodeType::Aztec => generators::aztec::generate_aztec(data)?,
67
68        // Phase 3: Legacy formats
69        BarcodeType::Code39 => generators::code39::generate_code39(data)?,
70        BarcodeType::ITF14 => generators::itf14::generate_itf14(data)?,
71        BarcodeType::Codabar => generators::codabar::generate_codabar(data)?
72    };
73
74    match format {
75        #[cfg(feature = "svg")]
76        ExportFormat::SVG => Ok(exporters::svg::export_svg(&barcode)?),
77        #[cfg(not(feature = "svg"))]
78        ExportFormat::SVG => Err(anyhow::anyhow!(
79            "SVG export not available - enable the 'svg' feature"
80        )),
81        #[cfg(feature = "png")]
82        ExportFormat::PNG => Ok(exporters::png::export_png(&barcode)?),
83        #[cfg(not(feature = "png"))]
84        ExportFormat::PNG => Err(anyhow::anyhow!(
85            "PNG export not available - enable the 'png' feature"
86        )),
87        #[cfg(feature = "pdf")]
88        ExportFormat::PDF => Ok(exporters::pdf::export_pdf(&barcode)?),
89        #[cfg(not(feature = "pdf"))]
90        ExportFormat::PDF => Err(anyhow::anyhow!(
91            "PDF export not available - enable the 'pdf' feature"
92        )),
93    }
94}

96/// Generate and save barcode to file
97///
98/// # Arguments
99///
100/// * `barcode_type` - The type of barcode to generate
101/// * `data` - The data to encode
102/// * `output_path` - Path where to save the file
103///
104/// # Examples
105///
106/// ```rust
107/// use quickcodes::{generate_to_file, BarcodeType};
108///
109/// fn main() -> Result<(), Box<dyn std::error::Error>> {
110///     // Generate QR Code and save as SVG
111///     generate_to_file(BarcodeType::QRCode, "Hello", "output.svg")?;
112///
113///     // Generate EAN-13 and save as PNG
114///     generate_to_file(BarcodeType::EAN13, "123456789012", "barcode.png")?;
115///     
116///     Ok(())
117/// }
118/// ```
119pub fn generate_to_file(
120    barcode_type: BarcodeType,
121    data: &str,
122    output_path: &str,
123) -> AnyhowResult<()> {
124    let format = ExportFormat::from_extension(output_path)?;
125    let barcode_data = generate(barcode_type, data, format)?;
126
127    std::fs::write(output_path, barcode_data)?;
128    Ok(())
129}

131/// Read barcode from image file
132///
133/// Automatically detects and decodes the first barcode found in the image.
134///
135/// # Arguments
136///
137/// * `image_path` - Path to the image file
138///
139/// # Returns
140///
141/// Returns the decoded barcode data and type information
142///
143/// # Examples
144///
145/// ```rust
146/// use quickcodes::read_from_file;
147///
148/// // This example would work if you had an actual barcode image
149/// // let result = read_from_file("barcode.png")?;
150/// // println!("Found {}: {}", result.barcode_type, result.data);
151/// ```
152pub fn read_from_file<P: AsRef<std::path::Path>>(_image_path: P) -> AnyhowResult<ReadResult> {
153    #[cfg(feature = "readers")]
154    {
155        Ok(readers_read_from_file(_image_path)?)
156    }
157
158    #[cfg(not(feature = "readers"))]
159    {
160        Err(anyhow::anyhow!(
161            "Reader functionality not available - enable the 'readers' feature"
162        ))
163    }
164}

166/// Read all barcodes from image file
167///
168/// Detects and decodes all barcodes found in the image.
169///
170/// # Arguments
171///
172/// * `image_path` - Path to the image file
173///
174/// # Returns
175///
176/// Returns a vector of all decoded barcodes
177pub fn read_all_from_file<P: AsRef<std::path::Path>>(
178    _image_path: P,
179) -> AnyhowResult<Vec<ReadResult>> {
180    #[cfg(feature = "readers")]

183    }
184
185    #[cfg(not(feature = "readers"))]

187        Err(anyhow::anyhow!(
188            "Reader functionality not available - enable the 'readers' feature"
189        ))
190    }
