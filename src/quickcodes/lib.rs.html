43/// Main generation function - unified API for all barcode types
44///
45/// # Arguments
46///
47/// * `barcode_type` - The type of barcode to generate
48/// * `data` - The data to encode
49/// * `format` - The output format (PNG, SVG)
50///
51/// # Returns
52///
53/// Returns the generated barcode as bytes in the specified format
54///
55/// # Examples
56///
57/// ```rust
58/// use quickcodes::{generate, BarcodeType, ExportFormat};
59///
60/// fn main() -> Result<(), Box<dyn std::error::Error>> {
61///     // Generate QR Code as SVG
62///     let svg_data = generate(BarcodeType::QRCode, "https://example.com", ExportFormat::SVG)?;
63///
64///     // Generate EAN-13 as PNG
65///     let png_data = generate(BarcodeType::EAN13, "123456789012", ExportFormat::PNG)?;
66///     
67///     Ok(())
68/// }
69/// ```
70pub fn generate(
71    barcode_type: BarcodeType,
72    data: &str,
73    format: ExportFormat,
74) -> AnyhowResult<Vec<u8>> {
75    let barcode = match barcode_type {
76        BarcodeType::QRCode => generators::qr::generate_qr(data)?,
77        BarcodeType::EAN13 => generators::ean13::generate_ean13(data)?,
78        BarcodeType::UPCA => generators::upc::generate_upc_a(data)?,
79        BarcodeType::Code128 => generators::code128::generate_code128(data)?,
80        _ => return Err(anyhow::anyhow!("Barcode type not yet implemented")),
81    };
82
83    match format {
84        #[cfg(feature = "svg")]
85        ExportFormat::SVG => Ok(exporters::svg::export_svg(&barcode)?),
86        #[cfg(not(feature = "svg"))]
87        ExportFormat::SVG => Err(anyhow::anyhow!(
88            "SVG export not available - enable the 'svg' feature"
89        )),
90        #[cfg(feature = "png")]
91        ExportFormat::PNG => Ok(exporters::png::export_png(&barcode)?),
92        #[cfg(not(feature = "png"))]
93        ExportFormat::PNG => Err(anyhow::anyhow!(
94            "PNG export not available - enable the 'png' feature"
95        )),
96        ExportFormat::PDF => Err(anyhow::anyhow!(
97            "PDF export not yet implemented - coming in Phase 2"
98        )),
99    }
100}

102/// Generate and save barcode to file
103///
104/// # Arguments
105///
106/// * `barcode_type` - The type of barcode to generate
107/// * `data` - The data to encode
108/// * `output_path` - Path where to save the file
109///
110/// # Examples
111///
112/// ```rust
113/// use quickcodes::{generate_to_file, BarcodeType};
114///
115/// fn main() -> Result<(), Box<dyn std::error::Error>> {
116///     // Generate QR Code and save as SVG
117///     generate_to_file(BarcodeType::QRCode, "Hello", "output.svg")?;
118///
119///     // Generate EAN-13 and save as PNG
120///     generate_to_file(BarcodeType::EAN13, "123456789012", "barcode.png")?;
121///     
122///     Ok(())
123/// }
124/// ```
125pub fn generate_to_file(
126    barcode_type: BarcodeType,
127    data: &str,
128    output_path: &str,
129) -> AnyhowResult<()> {
130    let format = ExportFormat::from_extension(output_path)?;
131    let barcode_data = generate(barcode_type, data, format)?;
132
133    std::fs::write(output_path, barcode_data)?;
134    Ok(())
135}
