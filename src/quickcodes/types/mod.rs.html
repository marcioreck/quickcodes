7/// Supported barcode types
8#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
9pub enum BarcodeType {
10    // 1D Barcodes
11    EAN13,
12    UPCA,
13    Code128,
14    Code39,
15    ITF14,
16    Codabar,
17
18    // 2D Codes
19    QRCode,
20    DataMatrix,
21    PDF417,
22    Aztec,
23}

25/// Export formats supported by QuickCodes
26#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
27pub enum ExportFormat {
28    PNG,
29    SVG,
30    PDF,
31}
32
33impl ExportFormat {
34    /// Determine export format from file extension
35    pub fn from_extension(path: &str) -> Result<Self> {
36        let path = Path::new(path);
37        match path.extension().and_then(|ext| ext.to_str()) {
38            Some("png") => Ok(ExportFormat::PNG),
39            Some("svg") => Ok(ExportFormat::SVG),
40            Some("pdf") => Ok(ExportFormat::PDF),
41            _ => Err(QuickCodesError::UnsupportedFormat(
42                "Unsupported file extension. Use .png, .svg, or .pdf".to_string(),
43            )),
44        }
45    }
46}

48/// QR Code error correction levels
49#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize, Default)]
50pub enum QRErrorCorrection {
51    Low, // ~7% recovery
52    #[default]
53    Medium, // ~15% recovery
54    Quartile, // ~25% recovery
55    High, // ~30% recovery
56}

58/// Configuration options for barcode generation
59#[derive(Debug, Clone, Serialize, Deserialize)]
60pub struct BarcodeConfig {
61    /// Width of the barcode in pixels (for raster formats)
62    pub width: Option<u32>,
63    /// Height of the barcode in pixels (for raster formats)
64    pub height: Option<u32>,
65    /// DPI for high-quality output
66    pub dpi: Option<u32>,
67    /// QR Code specific configuration
68    pub qr_config: QRConfig,
69    /// Whether to include human-readable text
70    pub include_text: bool,
71    /// Margin/quiet zone size
72    pub margin: u32,
73}
74
75impl Default for BarcodeConfig {
76    fn default() -> Self {
77        Self {
78            width: None,
79            height: None,
80            dpi: Some(300),
81            qr_config: QRConfig::default(),
82            include_text: true,
83            margin: 10,
84        }
85    }
86}
87
88/// QR Code specific configuration
89#[derive(Debug, Clone, Serialize, Deserialize)]
90pub struct QRConfig {
91    pub error_correction: QRErrorCorrection,
92    pub version: Option<u8>, // 1-40, None for auto
93}
94
95impl Default for QRConfig {
96    fn default() -> Self {
97        Self {
98            error_correction: QRErrorCorrection::Medium,
99            version: None,
100        }
101    }
102}

104/// Internal representation of a barcode
105#[derive(Debug, Clone)]
106pub struct Barcode {
107    /// The type of barcode
108    pub barcode_type: BarcodeType,
109    /// Original data that was encoded
110    pub data: String,
111    /// 2D matrix representation (for 2D codes) or 1D pattern (for 1D codes)
112    pub modules: BarcodeModules,
113    /// Configuration used to generate this barcode
114    pub config: BarcodeConfig,
115}
116
117/// Module representation for different barcode types
118#[derive(Debug, Clone)]
119pub enum BarcodeModules {
120    /// 1D barcode as a series of bars (true = black bar, false = white space)
121    Linear(Vec<bool>),
122    /// 2D barcode as a matrix (true = black module, false = white module)
123    Matrix(Vec<Vec<bool>>),
124}

126/// Errors that can occur during barcode generation or processing
127#[derive(Error, Debug)]
128pub enum QuickCodesError {
129    #[error("Invalid data: {0}")]
130    InvalidData(String),
131
132    #[error("Unsupported barcode type: {0:?}")]
133    UnsupportedBarcodeType(BarcodeType),
134
135    #[error("Unsupported export format: {0}")]
136    UnsupportedFormat(String),
137
138    #[error("Generation error: {0}")]
139    GenerationError(String),
140
141    #[error("Export error: {0}")]
142    ExportError(String),
143
144    #[error("IO error: {0}")]
145    IoError(#[from] std::io::Error),
146
147    #[error("Image processing error: {0}")]
148    ImageError(String),
149}
150
151/// Result type for QuickCodes operations
152pub type Result<T> = std::result::Result<T, QuickCodesError>;
